\section{Basic Functions}
\label{sec3::subsection::basics}

\begin{theorem}
\label{theorem::const}
    Let $c$ be an \alphacomputablereal{}. Then the constant function $f(x) = c$ is \WhileCCapproximable{}. 
\end{theorem}
\begin{proof}
Immediately follows from Lemma \ref{lemma::const_is_alpha_comp} and Equivalence Lemma \ref{lemma::equivalence_alpha_tracking_wcc_appx}.
\end{proof}

\begin{theorem}
\label{theorem::id}
    The identity function $id(x) = x$ is \WhileCCapproximable{}. 
\end{theorem}

\begin{proof}
Immediately follows from Lemma \ref{lemma::id_is_alpha_comp} and Equivalence Lemma \ref{lemma::equivalence_alpha_tracking_wcc_appx}.
\end{proof}

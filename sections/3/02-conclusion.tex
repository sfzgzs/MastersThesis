\section{Conclusion}
\label{sec::conclusion::elem-funcs-are-WCC-appxmable}
\begin{theorem}
\label{thm::elem_funcs_are_WCC_approximable}
All elementary functions are \WhileCC-approximable. 
\end{theorem}
\begin{proof}
        By induction on the structure of elementary functions, 
        theorems \ref{theorem::const} (constant function) and \ref{theorem::id} (identity function) prove the base cases, and 
        theorems
        \begin{itemize}
            \item 
        \ref{theorem::addition} (addition),
        \item \ref{theorem::multiplication} (multiplication),
        \item 
        \ref{theorem::division} (division), 
        \item \ref{theorem::log} (logarithm),
        \item 
        \ref{theorem::exp} (exponential),
        \item 
        \ref{theorem::sin} (sin), 
        \item 
        \ref{theorem::arcsine} (arcsin), and
        \item 
        \ref{theorem::odd-nth-root}, \ref{theorem::even-nth-root} (natural root) 
        \end{itemize}
        prove the induction step.
\end{proof}

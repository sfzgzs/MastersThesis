\begin{definition}[Acceptable Function]
\label{def::acceptable_function}
    A function $f:\Reals\to\Reals$ is \emph{\acceptable{}} if there exists a sequence $X$ where:
    \begin{enumerate}[(i)]
        \item $X$ is an \effectiveopenexhaustion{} for $\dom(f)$, and
        \item $f$ is \effectivelylocallyuniformlycontinuous{} w.r.t.\null{} $X$.
    \end{enumerate}

\end{definition}
